# CSV Import Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Create a mix task (`mix ancestry.import_csv family_echo "Family Name" path.csv`) that imports people and relationships from a FamilyEcho CSV export into a new family, using an adapter pattern.

**Architecture:** Three-layer design — `Ancestry.Import` (public API) delegates to `Ancestry.Import.CSV` (generic orchestrator with adapter behaviour) which calls `Ancestry.Import.CSV.FamilyEcho` (column-mapping adapter). A two-pass import: first creates all people, then creates relationships by looking up `external_id`.

**Tech Stack:** NimbleCSV (CSV parsing), Ecto (persistence), existing `Ancestry.People` and `Ancestry.Relationships` contexts.

```elixir
defmodule Ancestry.Import.CSV.FamilyEcho do
  @behaviour Ancestry.Import.CSV.Adapter

  @impl true
  def parse_person(row) do
    given = blank_to_nil(row["Given names"])
    surname = blank_to_nil(row["Surname now"])

    if is_nil(given) and is_nil(surname) do
      {:skip, "no name"}
    else
      attrs = %{
        external_id: "family_echo_#{row["ID"]}",
        given_name: given,
        surname: surname,
        surname_at_birth: blank_to_nil(row["Surname at birth"]),
        nickname: blank_to_nil(row["Nickname"]),
        title: blank_to_nil(row["Title"]),
        suffix: blank_to_nil(row["Suffix"]),
        gender: parse_gender(row["Gender"]),
        living: parse_living(row["Deceased"]),
        birth_year: parse_int(row["Birth year"]),
        birth_month: parse_int(row["Birth month"]),
        birth_day: parse_int(row["Birth day"]),
        death_year: parse_int(row["Death year"]),
        death_month: parse_int(row["Death month"]),
        death_day: parse_int(row["Death day"])
      }

      {:ok, attrs}
    end
  end

  @impl true
  def parse_relationships(row) do
    person_eid = "family_echo_#{row["ID"]}"

    []
    |> maybe_add_parent(row["Mother ID"], person_eid, "mother")
    |> maybe_add_parent(row["Father ID"], person_eid, "father")
    |> maybe_add_partner(row["Partner ID"], person_eid)
    |> maybe_add_ex_partners(row["Ex-partner IDs"], person_eid)
  end

  defp maybe_add_parent(rels, id, _person_eid, _role) when id in [nil, ""], do: rels

  defp maybe_add_parent(rels, id, person_eid, role) do
    [{:parent, "family_echo_#{id}", person_eid, %{role: role}} | rels]
  end

  defp maybe_add_partner(rels, id, _person_eid) when id in [nil, ""], do: rels

  defp maybe_add_partner(rels, id, person_eid) do
    [{:partner, person_eid, "family_echo_#{id}", %{}} | rels]
  end

  defp maybe_add_ex_partners(rels, ids, _person_eid) when ids in [nil, ""], do: rels

  defp maybe_add_ex_partners(rels, ids, person_eid) do
    ids
    |> String.split(",", trim: true)
    |> Enum.reduce(rels, fn id, acc ->
      [{:ex_partner, person_eid, "family_echo_#{String.trim(id)}", %{}} | acc]
    end)
  end

  defp blank_to_nil(nil), do: nil
  defp blank_to_nil(""), do: nil
  defp blank_to_nil(str), do: String.trim(str)

  defp parse_gender(nil), do: nil
  defp parse_gender(""), do: nil
  defp parse_gender("Female"), do: "female"
  defp parse_gender("Male"), do: "male"
  defp parse_gender(_), do: "other"

  defp parse_living("Y"), do: "no"
  defp parse_living(_), do: "yes"

  defp parse_int(nil), do: nil
  defp parse_int(""), do: nil

  defp parse_int(str) do
    case Integer.parse(String.trim(str)) do
      {n, _} -> n
      :error -> nil
    end
  end
end
```

```elixir
NimbleCSV.define(Ancestry.Import.CSVParser, separator: ",", escape: "\"")

defmodule Ancestry.Import.CSV do
  import Ecto.Query
  alias Ancestry.Repo
  alias Ancestry.Families
  alias Ancestry.People
  alias Ancestry.People.Person
  alias Ancestry.Relationships

  def import(adapter, family_name, csv_path) do
    with :ok <- validate_file(csv_path),
         {:ok, family} <- Families.create_family(%{name: family_name}) do
      rows = parse_csv(csv_path)

      {people_summary, _row_index} = import_people(adapter, family, rows)
      relationships_summary = import_relationships(adapter, rows)

      {:ok,
       %{
         family: family,
         people_created: people_summary.created,
         people_skipped: people_summary.skipped,
         people_errors: people_summary.errors,
         relationships_created: relationships_summary.created,
         relationships_skipped: relationships_summary.skipped,
         relationships_errors: relationships_summary.errors
       }}
    end
  end

  defp validate_file(path) do
    if File.exists?(path), do: :ok, else: {:error, "File not found: #{path}"}
  end

  defp parse_csv(path) do
    path
    |> File.read!()
    |> Ancestry.Import.CSVParser.parse_string(skip_headers: false)
    |> then(fn [headers | rows] ->
      headers = Enum.map(headers, &String.trim/1)

      Enum.map(rows, fn row ->
        headers
        |> Enum.zip(row)
        |> Map.new()
      end)
    end)
  end

  defp import_people(adapter, family, rows) do
    summary = %{created: 0, skipped: 0, errors: []}

    rows
    |> Enum.with_index(2)
    |> Enum.reduce({summary, 0}, fn {row, row_num}, {acc, idx} ->
      case adapter.parse_person(row) do
        {:ok, attrs} ->
          case People.create_person(family, attrs) do
            {:ok, _person} ->
              {%{acc | created: acc.created + 1}, idx + 1}

            {:error, changeset} ->
              error = "Row #{row_num}: #{inspect(errors_on(changeset))}"
              {%{acc | skipped: acc.skipped + 1, errors: [error | acc.errors]}, idx + 1}
          end

        {:skip, reason} ->
          error = "Row #{row_num}: #{reason}"
          {%{acc | skipped: acc.skipped + 1, errors: [error | acc.errors]}, idx + 1}
      end
    end)
  end

  defp import_relationships(adapter, rows) do
    summary = %{created: 0, skipped: 0, errors: []}

    rows
    |> Enum.with_index(2)
    |> Enum.reduce(summary, fn {row, row_num}, acc ->
      row
      |> adapter.parse_relationships()
      |> Enum.reduce(acc, fn {type, source_eid, target_eid, metadata}, rel_acc ->
        create_relationship(type, source_eid, target_eid, metadata, row_num, rel_acc)
      end)
    end)
  end

  defp create_relationship(type, source_eid, target_eid, metadata, row_num, acc) do
    source = Repo.get_by(Person, external_id: source_eid)
    target = Repo.get_by(Person, external_id: target_eid)

    cond do
      is_nil(source) ->
        error = "Row #{row_num} #{type}: person \"#{source_eid}\" not found"
        %{acc | skipped: acc.skipped + 1, errors: [error | acc.errors]}

      is_nil(target) ->
        error = "Row #{row_num} #{type}: person \"#{target_eid}\" not found"
        %{acc | skipped: acc.skipped + 1, errors: [error | acc.errors]}

      true ->
        type_string = Atom.to_string(type)

        case Relationships.create_relationship(source, target, type_string, metadata) do
          {:ok, _rel} ->
            %{acc | created: acc.created + 1}

          {:error, _reason} ->
            %{acc | skipped: acc.skipped + 1}
        end
    end
  end

  defp errors_on(changeset) do
    Ecto.Changeset.traverse_errors(changeset, fn {message, opts} ->
      Regex.replace(~r"%{(\w+)}", message, fn _, key ->
        opts |> Keyword.get(String.to_existing_atom(key), key) |> to_string()
      end)
    end)
  end
end
```

```elixir
defmodule Mix.Tasks.Ancestry.ImportCsv do
  @moduledoc "Import people and relationships from a CSV file into a new family."
  @shortdoc "Import people from CSV"

  use Mix.Task

  @impl Mix.Task
  def run(args) do
    Mix.Task.run("app.start")

    case args do
      [adapter_name, family_name, csv_path] ->
        adapter = String.to_atom(adapter_name)
        do_import(adapter, family_name, csv_path)

      _ ->
        print_usage()
    end
  end

  defp do_import(adapter, family_name, csv_path) do
    Mix.shell().info("Importing from #{csv_path} using #{adapter} adapter...")

    case Ancestry.Import.import_from_csv(adapter, family_name, csv_path) do
      {:ok, summary} ->
        Mix.shell().info("Created family: #{summary.family.name}\n")
        Mix.shell().info("Import complete!")
        Mix.shell().info("  People created:        #{summary.people_created}")
        Mix.shell().info("  People skipped:        #{summary.people_skipped}")
        Mix.shell().info("  Relationships created: #{summary.relationships_created}")
        Mix.shell().info("  Relationships skipped: #{summary.relationships_skipped}")

        if summary.people_errors != [] do
          Mix.shell().info("\nSkipped people:")

          Enum.each(Enum.reverse(summary.people_errors), fn error ->
            Mix.shell().info("  - #{error}")
          end)
        end

        if summary.relationships_errors != [] do
          Mix.shell().info("\nSkipped relationships:")

          Enum.each(Enum.reverse(summary.relationships_errors), fn error ->
            Mix.shell().info("  - #{error}")
          end)
        end

      {:error, reason} ->
        Mix.shell().error("Error: #{reason}")
    end
  end

  defp print_usage do
    available = Ancestry.Import.available_adapters() |> Enum.join(", ")

    Mix.shell().info("""
    Usage: mix ancestry.import_csv <adapter> <family_name> <csv_path>

    Available adapters: #{available}

    Example:
      mix ancestry.import_csv family_echo "Ferreira Family" family.csv
    """)
  end
end
```
