```elixir
# lib/ancestry/kinship.ex
defmodule Ancestry.Kinship do
  @moduledoc """
  Determines the kinship relationship between two people by finding their
  Most Recent Common Ancestor (MRCA) via bidirectional BFS.
  """

  alias Ancestry.Relationships
  alias Ancestry.People

  @max_depth 10

  defstruct [:relationship, :steps_a, :steps_b, :path, :mrca, :half?]

  @doc """
  Calculates the kinship between two people.

  Returns `{:ok, %Kinship{}}` with the relationship label, generation steps,
  and path from person_a through the MRCA to person_b.

  Returns `{:error, :same_person}` if both IDs are the same.
  Returns `{:error, :no_common_ancestor}` if no MRCA is found within #{@max_depth} generations.
  """
  def calculate(person_a_id, person_b_id) when person_a_id == person_b_id do
    {:error, :same_person}
  end

  def calculate(person_a_id, person_b_id) do
    # Check direct parent/child first (person_a is parent of person_b)
    case find_direct_relationship(person_a_id, person_b_id) do
      {:ok, result} ->
        {:ok, result}

      :not_direct ->
        # BFS: build ancestor maps and find intersection
        ancestors_a = build_ancestor_map(person_a_id)
        ancestors_b = build_ancestor_map(person_b_id)

        case find_mrca(ancestors_a, ancestors_b) do
          nil ->
            {:error, :no_common_ancestor}

          {mrca_id, steps_a, steps_b} ->
            half? = half_relationship?(person_a_id, person_b_id, mrca_id, steps_a, steps_b)
            relationship = classify(steps_a, steps_b, half?)
            path = build_path(person_a_id, person_b_id, mrca_id, ancestors_a, ancestors_b, steps_a, steps_b)

            {:ok,
             %__MODULE__{
               relationship: relationship,
               steps_a: steps_a,
               steps_b: steps_b,
               path: path,
               mrca: People.get_person!(mrca_id),
               half?: half?
             }}
        end
    end
  end

  # --- Direct relationship check ---

  defp find_direct_relationship(person_a_id, person_b_id) do
    # Check if A is ancestor of B
    case find_ancestor_path(person_a_id, person_b_id, 0) do
      {:found, depth, path_ids} ->
        person_a = People.get_person!(person_a_id)
        relationship = classify(0, depth, false)
        path = build_direct_path(path_ids, 0, depth)
        {:ok, %__MODULE__{relationship: relationship, steps_a: 0, steps_b: depth, path: path, mrca: person_a, half?: false}}

      :not_found ->
        # Check if B is ancestor of A
        case find_ancestor_path(person_b_id, person_a_id, 0) do
          {:found, depth, path_ids} ->
            person_a = People.get_person!(person_a_id)
            relationship = classify(depth, 0, false)
            path = build_direct_path(Enum.reverse(path_ids), depth, 0)
            {:ok, %__MODULE__{relationship: relationship, steps_a: depth, steps_b: 0, path: path, mrca: People.get_person!(person_b_id), half?: false}}

          :not_found ->
            :not_direct
        end
    end
  end

  defp find_ancestor_path(_ancestor_id, _descendant_id, depth) when depth > @max_depth, do: :not_found

  defp find_ancestor_path(ancestor_id, descendant_id, depth) do
    if ancestor_id == descendant_id do
      {:found, depth, [ancestor_id]}
    else
      parents = Relationships.get_parents(descendant_id)

      Enum.reduce_while(parents, :not_found, fn {parent, _rel}, _acc ->
        case find_ancestor_path(ancestor_id, parent.id, depth) do
          {:found, d, path} -> {:halt, {:found, d + 1, path ++ [descendant_id]}}
          :not_found -> {:cont, :not_found}
        end
      end)
    end
  end

  # --- Ancestor map building (BFS) ---

  defp build_ancestor_map(person_id) do
    bfs(
      [{person_id, 0, [person_id]}],
      %{person_id => {0, [person_id]}},
      0
    )
  end

  defp bfs([], visited, _depth), do: visited
  defp bfs(_queue, visited, depth) when depth >= @max_depth, do: visited

  defp bfs(queue, visited, depth) do
    next_queue =
      Enum.flat_map(queue, fn {pid, gen, path} ->
        parents = Relationships.get_parents(pid)

        Enum.flat_map(parents, fn {parent, _rel} ->
          if Map.has_key?(visited, parent.id) do
            []
          else
            [{parent.id, gen + 1, path ++ [parent.id]}]
          end
        end)
      end)

    new_visited =
      Enum.reduce(next_queue, visited, fn {pid, gen, path}, acc ->
        Map.put_new(acc, pid, {gen, path})
      end)

    bfs(next_queue, new_visited, depth + 1)
  end

  # --- MRCA finding ---

  defp find_mrca(ancestors_a, ancestors_b) do
    common =
      ancestors_a
      |> Map.keys()
      |> Enum.filter(&Map.has_key?(ancestors_b, &1))
      |> Enum.map(fn id ->
        {gen_a, _} = Map.get(ancestors_a, id)
        {gen_b, _} = Map.get(ancestors_b, id)
        {id, gen_a, gen_b}
      end)
      |> Enum.sort_by(fn {_id, ga, gb} -> ga + gb end)

    List.first(common)
  end

  # --- Half-relationship detection ---

  defp half_relationship?(_a_id, _b_id, _mrca_id, steps_a, steps_b) when steps_a == 0 or steps_b == 0 do
    false
  end

  defp half_relationship?(person_a_id, person_b_id, mrca_id, steps_a, steps_b) do
    # For siblings (steps 1,1): check if they share both parents or just one
    # For cousins: check if the MRCA's children on each path share both parents with their siblings
    if steps_a == 1 and steps_b == 1 do
      # Sibling case: count shared parents
      parents_a = Relationships.get_parents(person_a_id) |> Enum.map(fn {p, _} -> p.id end) |> MapSet.new()
      parents_b = Relationships.get_parents(person_b_id) |> Enum.map(fn {p, _} -> p.id end) |> MapSet.new()
      MapSet.size(MapSet.intersection(parents_a, parents_b)) == 1
    else
      # For cousins and further: check if MRCA is a single shared ancestor
      # rather than a couple (both parents shared)
      # Simple heuristic: if MRCA appears only once (not as part of a couple), it's half
      parents_a = get_ancestor_at_depth(person_a_id, steps_a - 1)
      parents_b = get_ancestor_at_depth(person_b_id, steps_b - 1)

      # The children of MRCA on path A and path B
      child_on_a_path = parents_a
      child_on_b_path = parents_b

      if child_on_a_path && child_on_b_path && child_on_a_path != child_on_b_path do
        parents_of_a_child = Relationships.get_parents(child_on_a_path) |> Enum.map(fn {p, _} -> p.id end) |> MapSet.new()
        parents_of_b_child = Relationships.get_parents(child_on_b_path) |> Enum.map(fn {p, _} -> p.id end) |> MapSet.new()
        MapSet.size(MapSet.intersection(parents_of_a_child, parents_of_b_child)) == 1
      else
        false
      end
    end
  end

  defp get_ancestor_at_depth(person_id, 0), do: person_id

  defp get_ancestor_at_depth(person_id, depth) do
    case Relationships.get_parents(person_id) do
      [{parent, _} | _] -> get_ancestor_at_depth(parent.id, depth - 1)
      [] -> nil
    end
  end

  # --- Relationship classification ---

  defp classify(0, 0, _half?), do: "Self"
  defp classify(0, 1, _half?), do: "Parent"
  defp classify(1, 0, _half?), do: "Child"
  defp classify(0, 2, _half?), do: "Grandparent"
  defp classify(2, 0, _half?), do: "Grandchild"
  defp classify(0, 3, _half?), do: "Great-Grandparent"
  defp classify(3, 0, _half?), do: "Great-Grandchild"

  defp classify(0, n, _half?) when n > 3 do
    greats = n - 2
    prefix = if greats == 1, do: "Great-", else: "#{ordinal(greats - 1)} Great-"
    "#{prefix}Grandparent"
  end

  defp classify(n, 0, _half?) when n > 3 do
    greats = n - 2
    prefix = if greats == 1, do: "Great-", else: "#{ordinal(greats - 1)} Great-"
    "#{prefix}Grandchild"
  end

  defp classify(1, 1, true), do: "Half-Sibling"
  defp classify(1, 1, false), do: "Sibling"

  defp classify(1, 2, _half?), do: "Uncle/Aunt"
  defp classify(2, 1, _half?), do: "Niece/Nephew"
  defp classify(1, 3, _half?), do: "Great-Uncle/Aunt"
  defp classify(3, 1, _half?), do: "Great-Niece/Nephew"
  defp classify(1, n, _half?) when n > 3, do: "#{ordinal(n - 2)} Great-Uncle/Aunt"
  defp classify(n, 1, _half?) when n > 3, do: "#{ordinal(n - 2)} Great-Niece/Nephew"

  defp classify(steps_a, steps_b, half?) do
    degree = min(steps_a, steps_b) - 1
    removed = abs(steps_a - steps_b)
    prefix = if half?, do: "Half-", else: ""

    base = "#{prefix}#{ordinal(degree)} Cousin"

    case removed do
      0 -> base
      1 -> "#{base}, Once Removed"
      2 -> "#{base}, Twice Removed"
      n -> "#{base}, #{n}x Removed"
    end
  end

  defp ordinal(1), do: "First"
  defp ordinal(2), do: "Second"
  defp ordinal(3), do: "Third"
  defp ordinal(4), do: "Fourth"
  defp ordinal(5), do: "Fifth"
  defp ordinal(n), do: "#{n}th"

  # --- Path building ---

  defp build_path(person_a_id, person_b_id, mrca_id, ancestors_a, ancestors_b, steps_a, steps_b) do
    {_, path_a_ids} = Map.get(ancestors_a, mrca_id)
    {_, path_b_ids} = Map.get(ancestors_b, mrca_id)

    # path_a goes from person_a up to mrca
    # path_b goes from person_b up to mrca
    # We want: person_a -> ... -> mrca -> ... -> person_b
    path_up = path_a_ids
    path_down = path_b_ids |> Enum.reverse() |> tl()

    full_path_ids = path_up ++ path_down

    full_path_ids
    |> Enum.with_index()
    |> Enum.map(fn {pid, idx} ->
      person = People.get_person!(pid)
      label = label_for_position(idx, steps_a, steps_b, length(full_path_ids))
      %{person: person, label: label}
    end)
  end

  defp build_direct_path(path_ids, steps_a, steps_b) do
    total = length(path_ids)

    path_ids
    |> Enum.with_index()
    |> Enum.map(fn {pid, idx} ->
      person = People.get_person!(pid)
      label = label_for_position(idx, steps_a, steps_b, total)
      %{person: person, label: label}
    end)
  end

  defp label_for_position(0, _steps_a, _steps_b, _total), do: "Self"

  defp label_for_position(idx, steps_a, steps_b, total) do
    # Path goes: person_a (0) -> ... -> mrca (steps_a) -> ... -> person_b (total-1)
    cond do
      idx < steps_a ->
        # Going up from A to MRCA
        classify(0, idx, false)

      idx == steps_a ->
        # At MRCA
        classify(0, steps_a, false)

      true ->
        # Going down from MRCA to B
        down_steps = idx - steps_a
        classify(steps_a, down_steps + steps_a, false) |> label_from_b_perspective(idx, steps_a, steps_b, total)
    end
  end

  # On the descent from MRCA to B, label each person relative to A
  defp label_from_b_perspective(_base_label, idx, steps_a, _steps_b, total) do
    down_from_mrca = idx - steps_a
    up_to_mrca = steps_a

    classify(up_to_mrca, down_from_mrca, false)
  end
end
```

     |> assign(:search_b, "")
     |> assign(:filtered_a, [])
     |> assign(:filtered_b, [])
     |> assign(:dropdown_a, false)
     |> assign(:dropdown_b, false)}
  end

  @impl true
  def handle_params(params, _url, socket) do
    people = socket.assigns.people

    person_a =
      case params do
        %{"person_a" => id} -> Enum.find(people, &(&1.id == String.to_integer(id)))
        _ -> nil
      end

    person_b =
      case params do
        %{"person_b" => id} -> Enum.find(people, &(&1.id == String.to_integer(id)))
        _ -> nil
      end

    result = maybe_calculate(person_a, person_b)

    {:noreply,
     socket
     |> assign(:person_a, person_a)
     |> assign(:person_b, person_b)
     |> assign(:result, result)}
  end

  @impl true
  def handle_event("toggle_dropdown_a", _, socket) do
    {:noreply,
     socket
     |> assign(:dropdown_a, !socket.assigns.dropdown_a)
     |> assign(:dropdown_b, false)
     |> assign(:search_a, "")
     |> assign(:filtered_a, socket.assigns.people)}
  end

  def handle_event("toggle_dropdown_b", _, socket) do
    {:noreply,
     socket
     |> assign(:dropdown_b, !socket.assigns.dropdown_b)
     |> assign(:dropdown_a, false)
     |> assign(:search_b, "")
     |> assign(:filtered_b, socket.assigns.people)}
  end

  def handle_event("close_dropdowns", _, socket) do
    {:noreply,
     socket
     |> assign(:dropdown_a, false)
     |> assign(:dropdown_b, false)}
  end

  def handle_event("filter_a", %{"value" => query}, socket) do
    filtered = filter_people(socket.assigns.people, query)
    {:noreply, socket |> assign(:search_a, query) |> assign(:filtered_a, filtered)}
  end

  def handle_event("filter_b", %{"value" => query}, socket) do
    filtered = filter_people(socket.assigns.people, query)
    {:noreply, socket |> assign(:search_b, query) |> assign(:filtered_b, filtered)}
  end

  def handle_event("select_person_a", %{"id" => id}, socket) do
    person = Enum.find(socket.assigns.people, &(&1.id == String.to_integer(id)))
    result = maybe_calculate(person, socket.assigns.person_b)

    {:noreply,
     socket
     |> assign(:person_a, person)
     |> assign(:result, result)
     |> assign(:dropdown_a, false)}
  end

  def handle_event("select_person_b", %{"id" => id}, socket) do
    person = Enum.find(socket.assigns.people, &(&1.id == String.to_integer(id)))
    result = maybe_calculate(socket.assigns.person_a, person)

    {:noreply,
     socket
     |> assign(:person_b, person)
     |> assign(:result, result)
     |> assign(:dropdown_b, false)}
  end

  def handle_event("clear_a", _, socket) do
    {:noreply, socket |> assign(:person_a, nil) |> assign(:result, nil)}
  end

  def handle_event("clear_b", _, socket) do
    {:noreply, socket |> assign(:person_b, nil) |> assign(:result, nil)}
  end

  def handle_event("swap", _, socket) do
    a = socket.assigns.person_a
    b = socket.assigns.person_b
    result = maybe_calculate(b, a)

    {:noreply,
     socket
     |> assign(:person_a, b)
     |> assign(:person_b, a)
     |> assign(:result, result)}
  end

  defp maybe_calculate(nil, _), do: nil
  defp maybe_calculate(_, nil), do: nil

  defp maybe_calculate(person_a, person_b) do
    case Kinship.calculate(person_a.id, person_b.id) do
      {:ok, result} -> result
      {:error, _} = error -> error
    end
  end

  defp filter_people(people, query) do
    q = String.downcase(String.trim(query))

    if q == "" do
      people
    else
      Enum.filter(people, fn p ->
        name = String.downcase(Person.display_name(p))
        String.contains?(name, q)
      end)
    end
  end
end

defmodule Web.UserFlows.CalculatingKinshipTest do
  use Web.E2ECase

  # Given a family with known relationships (grandparent, two parents, two cousins)
  # When the user navigates to the kinship page
  # And selects two cousins
  # Then the relationship "First Cousin" is displayed with a path visualization
  #
  # When the user swaps the people
  # Then the labels update correctly
  #
  # When the user clears a selection
  # Then the result disappears
  #
  # When the user selects two unrelated people
  # Then "No common ancestor found" is shown

  setup do
    family = insert(:family, name: "Kinship Test Family")
    grandparent = insert(:person, given_name: "Alice", surname: "Smith")
    parent_a = insert(:person, given_name: "Bob", surname: "Smith")
    parent_b = insert(:person, given_name: "Carol", surname: "Smith")
    cousin_a = insert(:person, given_name: "Dan", surname: "Smith")
    cousin_b = insert(:person, given_name: "Eve", surname: "Smith")
    unrelated = insert(:person, given_name: "Zane", surname: "Unrelated")

    # Add all to family
    for person <- [grandparent, parent_a, parent_b, cousin_a, cousin_b, unrelated] do
      Ancestry.People.add_to_family(person, family)
    end

    # Create relationships
    Ancestry.Relationships.create_relationship(grandparent, parent_a, "parent", %{role: "mother"})
    Ancestry.Relationships.create_relationship(grandparent, parent_b, "parent", %{role: "mother"})
    Ancestry.Relationships.create_relationship(parent_a, cousin_a, "parent", %{role: "father"})
    Ancestry.Relationships.create_relationship(parent_b, cousin_b, "parent", %{role: "mother"})

    %{
      family: family,
      grandparent: grandparent,
      parent_a: parent_a,
      parent_b: parent_b,
      cousin_a: cousin_a,
      cousin_b: cousin_b,
      unrelated: unrelated
    }
  end

  test "calculating kinship between two cousins", %{
    family: family,
    cousin_a: cousin_a,
    cousin_b: cousin_b,
    unrelated: unrelated
  } do
    # Navigate to family show page
    conn = visit(~p"/families/#{family.id}")
